<!-- Description: A circular ring with top point A and bottom point C. Two equal loads W act symmetrically downward/inward at angles on either side of the top. The included angles are labeled $\theta$ and $\pi+\theta$. A vertical upward force at C is labeled $2W \cos \theta$. -->

$$
LT_M=-WR\sin (x-\theta)(x-\theta)^0
$$

$$
LT_N=-W\sin (x-\theta)(x-\theta)^0
$$

$$
LT_V=-W\cos (x-\theta)(x-\theta)^0
$$

$$
M_A=-\frac{WR}{\pi}[s(\pi-\theta)-k_2(1+c)]
$$

$$
M_C=-\frac{WR}{\pi}[s\theta-k_2(1+c)]
$$

$$
N_A=-\frac{W}{\pi}s(\pi-\theta)
$$

$$
V_A=0
$$

$$
\Delta D_H=
\begin{cases}
\left[-\frac{WR^3}{EI}\left\{k_1\left(\frac{\theta s}{2}-c\right)+2k_2c-\frac{2k_2^2(1+c)}{\pi}\right\}\right], & \text{if } \theta\le \frac{\pi}{2} \\[6pt]
\left[-\frac{WR^3}{EI}\left\{k_1\frac{s(\pi-\theta)}{2}-\frac{2k_2^2(1+c)}{\pi}\right\}\right], & \text{if } \theta\ge \frac{\pi}{2}
\end{cases}
$$

$$
\Delta D_V=\frac{WR^3}{EI}\left[\frac{k_1(s-\pi c+\theta c)}{2}-k_2s+\frac{2k_2^2(1+c)}{\pi}\right]
$$

$$
\Delta L=
\begin{cases}
\frac{WR^3}{2EI}\left[k_1\left(\frac{\theta s}{\pi}-\frac{\pi c}{2}\right)-k_2(1-c)+\frac{2k_2^2(1+c)}{\pi}\right], & \text{if } \theta\le \frac{\pi}{2} \\[8pt]
\frac{WR^3}{2EI}\left[k_1\left(\frac{\theta s}{\pi}-\pi c+\theta c\right)+k_2(1+c-2s)+\frac{2k_2^2(1+c)}{\pi}\right], & \text{if } \theta\ge \frac{\pi}{2}
\end{cases}
$$

$$
\Delta L_W=\frac{WR^3}{EI}\left\{k_1\frac{s-s^3(1-\theta/\pi)-c(\pi-\theta)}{2}+k_2\left[\frac{\theta s(1+c)}{\pi}-s+\frac{k_2^2(1+c)^2}{\pi}\right]\right\}
$$

$$
\Delta L_{WH}=-\frac{WR^3}{EI\pi}[k_1s(\pi-\theta)(\theta-sc)+2\theta ck_2(1+c)-2sk_2^2(1+c)]
$$

$$
\Delta\psi=\frac{WR^2}{EI\pi}[\pi s^2-\theta(1+c+s^2)]k_2
$$

$$
\text{Max }+M=M_C \quad \text{if } \theta\le 60^\circ
$$

Max $+M$ occurs at the load if $\theta>60^\circ$ where

$$
M_\theta=\frac{WR}{\pi}[k_2(1+c)-sc(\pi-\theta)]
$$

$$
\text{Max }-M=
\begin{cases}
M_C, & \text{if } \theta\ge 90^\circ \\
M_A, & \text{if } 60^\circ\le \theta\le 90^\circ
\end{cases}
$$

Max $-M$ occurs at an angular position

$$
x_1=\tan^{-1}\frac{-\pi c}{\theta s} \quad \text{if } \theta\le 60^\circ
$$

If $\alpha=\beta=0$, $M=K_MWR$, $N=K_NW$, $\Delta D=K_{\Delta D}WR^3/EI$,

$$
\Delta\psi=K_{\Delta\psi}WR^2/EI,
$$

etc.
