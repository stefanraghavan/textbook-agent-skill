$$
V_1=\frac{Pt_1K_{V1}}{\pi R_1^2}, \qquad
M_1=\frac{Pt_1^2K_{M1}}{\pi R_1^2}, \qquad
N_1=\frac{P}{2\pi R_1}
$$

$$
\Delta R_A=\frac{Pt_1}{E_1\pi R_1^2}\left(LT_{A1}-K_{V1}C_{AA1}-K_{M1}C_{AB1}\right)
$$

$$
\psi_A=\frac{P}{E_1\pi R_1^2}\left(K_{V1}C_{AB1}+K_{M1}C_{BB1}\right)
$$
