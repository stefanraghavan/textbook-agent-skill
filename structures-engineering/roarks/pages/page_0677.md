V_1=\delta_1\omega^2R_1t_1^2K_{V1}, \qquad M_1=\delta_1\omega^2R_1t_1^3K_{M1}
$$

$$
N_1=-V_1\sin\alpha_1
$$

$$
\Delta R_A=\frac{\delta_1\omega^2R_1t_1^2}{E_1}(LT_{A1}-K_{V1}C_{AA1}-K_{M1}C_{AB1})
$$

$$
\psi_A=\frac{\delta_1\omega^2R_1t_1}{E_1}(-LT_{B1}+K_{V1}C_{AB1}+K_{M1}C_{BB1})
$$
