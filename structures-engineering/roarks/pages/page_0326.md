$$
LT_M=\frac{-wR^2}{2}(c-u)^2(x-\theta)^0
$$

$$
LT_N=wRu(c-u)(x-\theta)^0
$$

$$
LT_V=-wRz(c-u)(x-\theta)^0
$$

$$
M_A=\frac{-wR^2}{4\pi}\left[(\pi-\theta)(4c+2s^2-1)+s\left(4-\frac{4s^2}{3}-c\right)-2k_2(\pi-\theta+sc)\right]
$$

$$
M_C=\frac{-wR^2}{4\pi}\left[3\pi+\theta+4\theta c-2\theta s^2-4s-sc+\frac{4s^3}{3}-2k_2(\pi-\theta+sc)\right]
$$

$$
N_A=\frac{-wR}{\pi}\left(\pi c+s-\theta c-\frac{s^3}{3}\right)
$$

$$
V_A=0
$$

$$
\Delta D_H=
\begin{cases}
\displaystyle
\frac{-wR^4}{6EI\pi}\left[\pi k_1(s^3+3\theta c+4-3s)+3k_2(\pi-\theta+2\theta c^2-sc)-6k_2^2(\pi-\theta+sc)\right], & \text{for } \theta\le \frac{\pi}{2} \\[1.2em]
\displaystyle
\frac{-wR^4}{2EI\pi}\left\{\pi k_1\left[c(\pi-\theta)+s-\frac{s^3}{3}\right]+k_2[(\pi-\theta)(2s^2-1)-sc]-2k_2^2(\pi-\theta+sc)\right\}, & \text{for } \theta\ge \frac{\pi}{2}
\end{cases}
$$

$$
\Delta D_V=\frac{wR^4}{3EI\pi}\left\{\pi k_1(2-c^3+3c)+3k_2[2\theta s^2-\theta+sc-\pi(1+2c+s^3)]+6k_2^2(\pi-\theta+sc)\right\}
$$

$$
\Delta L=
\begin{cases}
\displaystyle
\frac{wR^4}{12EI\pi}\left\{1.5\pi(\theta-2\theta s^2-sc)+2k_1(2\pi+s^3+3\theta c-3s)+3k_2(sc+\theta\pi+2\theta s^2-3\pi-\theta-\pi sc)+6k_2^2(\pi-\theta+sc)\right\}, & \text{for } \theta\le \frac{\pi}{2} \\[1.2em]
\displaystyle
\frac{wR^4}{12EI\pi}\left\{1.5\pi[(\pi-\theta)(1-2s^2)+sc]+2k_1(2\pi+s^3+3\theta c-3s-\pi c^3)+3k_2[(\pi+1)(\pi-\theta+sc)+2\theta s^2-4\pi(1+c)]+6k_2^2(\pi-\theta+sc)\right\}, & \text{for } \theta\ge \frac{\pi}{2}
\end{cases}
$$

Max $+\,M$ occurs at an angular position $x_1$ where $x_1>\theta$, $x_1>90^\circ$, and

$$
x_1=\cos^{-1}\frac{s^3/3+\theta c-s}{\pi}
$$

Max $-\,M=M_C$
