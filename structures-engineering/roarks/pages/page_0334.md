$$
M_A=\frac{WR}{2\pi}\left[n^2-s^2-(\pi-\phi)n+(\pi-\theta)s-k_2(c-m)\right]
$$

$$
N_A=\frac{W}{2\pi}(n^2-s^2)
$$

$$
V_A=\frac{W}{2\pi}(\theta-\phi+s-n+sc-nm)
$$

If $\alpha=\beta=0$, $M=K_MWR$, $N=K_NW$, $V=K_VW$

$$
v=\frac{W}{2\pi R}\left(\sin\phi-\sin\theta\right)
$$

$$
LT_M=-\frac{WR}{2\pi}(n-s)(x-z)
$$

$$
\qquad +WR(z-s)\langle x-\theta\rangle^0
$$

$$
\qquad -WR(z-n)\langle x-\phi\rangle^0
$$

$$
LT_N=\frac{W}{2\pi}(n-s)z
$$

$$
\qquad +Wz\langle x-\theta\rangle^0
$$

$$
\qquad -Wz\langle x-\phi\rangle^0
$$

$$
LT_V=-\frac{W}{2\pi}(n-s)(1-u)
$$

$$
\qquad +Wu\langle x-\theta\rangle^0
$$

$$
\qquad -Wu\langle x-\phi\rangle^0
$$
