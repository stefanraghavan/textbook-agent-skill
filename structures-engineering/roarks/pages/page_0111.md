$$
\varepsilon_A=\kappa\{(1-\nu_0K_{tA})(3-K_{tB}-K_{tC}-K_{tB}K_{tC})\hat{\varepsilon}_A
-2K_{tA}[(1-\nu_0K_{tB})(1-K_{tC})\hat{\varepsilon}_B
+(1-\nu_0K_{tC})(1-K_{tB})\hat{\varepsilon}_C]\}
$$

$$
\varepsilon_B=\kappa\{(1-\nu_0K_{tB})(3-K_{tC}-K_{tA}-K_{tC}K_{tA})\hat{\varepsilon}_B
-2K_{tB}[(1-\nu_0K_{tC})(1-K_{tA})\hat{\varepsilon}_C
+(1-\nu_0K_{tA})(1-K_{tC})\hat{\varepsilon}_A]\}
$$

$$
\varepsilon_C=\kappa\{(1-\nu_0K_{tC})(3-K_{tA}-K_{tB}-K_{tA}K_{tB})\hat{\varepsilon}_C
-2K_{tC}[(1-\nu_0K_{tA})(1-K_{tB})\hat{\varepsilon}_A
+(1-\nu_0K_{tB})(1-K_{tA})\hat{\varepsilon}_B]\}
$$

where

$$
\kappa=(3K_{tA}K_{tB}K_{tC}-K_{tA}K_{tB}-K_{tB}K_{tC}-K_{tA}K_{tC}-K_{tA}-K_{tB}-K_{tC}+3)^{-1}
$$
