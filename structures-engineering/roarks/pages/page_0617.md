$$
F_{10A} = F_{7A} = F_{6A} = 1 - \frac{2.652}{k_A} + \frac{1.641}{k_A^2} - \frac{0.290}{k_A^3} - \frac{2.211}{k_A^4}

$$

$$
F_{8A} = F_{5A} = 1 - \frac{3.359}{k_A} + \frac{7.266}{k_A^2} - \frac{10.068}{k_A^3} + \frac{5.787}{k_A^4}
$$

$$
F_{9A} = C_1 = F_{5A} + \frac{2\sqrt{2}\nu}{k_A} F_{2A}

$$
N_1 = Q_A \sin\alpha \left(\frac{k_A}{k}\right)^{5/2} \frac{e^{-\mu}}{C_1} (F_9 \cos\mu - F_{10} \sin\mu)
$$

$$
N_2 = Q_A \sin\alpha \left(\frac{k_A}{k}\right)^{3/2} \frac{k_A}{\sqrt{2}} \frac{e^{-\mu}}{C_1} (F_{11} \cos\mu + F_{12} \sin\mu)
$$

$$
M_1 = Q_A \sin\alpha \left(\frac{k_A}{k}\right)^{3/2} \frac{k_A t}{\sqrt{2}\beta}\frac{e^{-\mu}}{C_1}
\left[
- \left(F_{12} - \frac{\sqrt{2}\nu}{k} F_{10}\right)\cos\mu
+ \left(F_{11} + \frac{\sqrt{2}\nu}{k} F_9\right)\sin\mu
\right]
$$

$$
M_2 = Q_A \nu \sin\alpha \left(\frac{k_A}{k}\right)^{3/2} \frac{k_A t}{\sqrt{2}\beta}\frac{e^{-\mu}}{C_1}
\left[
- \left(F_{12} - \frac{\sqrt{2}}{\nu k} F_{10}\right)\cos\mu
+ \left(F_{11} + \frac{\sqrt{2}}{\nu k} F_9\right)\sin\mu
\right]
$$

$$
\Delta R = \frac{Q_A R}{Et}\sin\alpha \left(\frac{k_A}{k}\right)^{3/2} \frac{k_A}{\sqrt{2}} \frac{e^{-\mu}}{C_1}
\left[
\left(F_{11} - \frac{\sqrt{2}\nu}{k} F_9\right)\cos\mu
+ \left(F_{12} + \frac{\sqrt{2}\nu}{k} F_{10}\right)\sin\mu
\right]
$$

$$
\psi = \frac{Q_A R_A \beta}{Et^2} \left(\frac{k_A}{k}\right)^{1/2} \frac{e^{-\mu}}{C_1} (F_{10} \cos\mu + F_9 \sin\mu)
$$

At the loaded end where $R = R_A$,

$$
N_{1A} = Q_A \sin\alpha,
\qquad
N_{2A} = Q_A \sin\alpha \frac{k_A}{\sqrt{2}C_1} \left(F_{4A} + \frac{\sqrt{2}\nu}{k_A} F_{8A}\right)
$$

$$
M_{1A} = 0,
\qquad
M_{2A} = Q_A \sin\alpha (1-\nu^2)\frac{t}{\beta C_1} F_{10A}
$$

$$
\Delta R_A = \frac{Q_A R_A}{Et}\sin\alpha \frac{k_A}{\sqrt{2}C_1}
\left(F_{4A} - \frac{4\nu^2}{k_A^2} F_{2A}\right),
\qquad
\psi_A = \frac{Q_A R_A \beta}{Et^2 C_1} F_{10A}
$$
