$$
LT_M = WR(z-s)\langle x-\theta \rangle^0
$$

$$
LT_N = Wz\langle x-\theta \rangle^0
$$

$$
LT_V = Wu\langle x-\theta \rangle^0
$$

$$
M_A = -\frac{WR}{\pi}[s(s-\pi+\theta)+k_2(1+c)]
$$

$$
M_C = -\frac{WR}{\pi}[s\theta-s^2+k_2(1+c)]
$$

$$
N_A = -\frac{W}{\pi}s^2
$$

$$
V_A = 0
$$

$$
\Delta D_H =
\begin{cases}
-\dfrac{WR^3}{EI\pi}\left[\pi k_1\left(1-\dfrac{s^2}{2}\right)-2k_2(\pi-\theta s)+2k_2^2(1+c)\right] & \text{if } \theta \le \dfrac{\pi}{2} \\[8pt]
-\dfrac{WR^3}{EI\pi}\left[\dfrac{\pi k_1 s^2}{2}-2sk_2(\pi-\theta)+2k_2^2(1+c)\right] & \text{if } \theta \ge \dfrac{\pi}{2}
\end{cases}
$$

$$
\Delta D_V = \frac{WR^3}{EI\pi}\left[\frac{\pi k_1(\pi-\theta-sc)}{2}+k_2s(\pi-2\theta)-2k_2^2(1+c)\right]
$$

$$
\Delta L =
\begin{cases}
\dfrac{WR^3}{2EI}\left[\theta s+k_1\left(\dfrac{\pi}{2}-\dfrac{s^2}{\pi}\right)-k_2\left(1-c+\dfrac{2\theta s}{\pi}\right)-\dfrac{2k_2^2(1+c)}{\pi}\right] & \text{if } \theta \le \dfrac{\pi}{2} \\[10pt]
\dfrac{WR^3}{2EI}\left[s(\pi-\theta)+k_1\left(\pi-\theta-sc-\dfrac{s^2}{\pi}\right)-k_2\left(1+c+\dfrac{2\theta s}{\pi}\right)-\dfrac{2k_2^2(1+c)}{\pi}\right] & \text{if } \theta \ge \dfrac{\pi}{2}
\end{cases}
$$

$$
\Delta L_W = \frac{WR^3}{EI\pi}\left[\theta s^2(\pi-\theta)+\frac{\pi k_1(\pi-\theta-sc-s^4/\pi)}{2}+k_2s(\pi c-2\theta-2\theta c)-k_2^2(1+c)^2\right]
$$

$$
\Delta L_{WH} = -\frac{WR^3}{EI\pi}\left[2\theta sc(\pi-\theta)+k_1s^2(\theta-sc)-2k_2(\pi s^2-\theta s^2+\theta c+\theta c^2)+2k_2^2s(1+c)\right]
$$

$$
\Delta \psi = \frac{WR^2}{EI\pi}\left[-\theta s(\pi-\theta)+k_2(\theta+\theta c+s^3)\right]
$$

Max $+M$ occurs at an angular position $x_1=\tan^{-1}\frac{-\pi}{s^2}$ if $\theta < 106.3^\circ$
